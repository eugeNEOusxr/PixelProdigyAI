┌──────────────────────────────────────────────────────────────────────────────┐
│                    📐 3D Space Calculations                                   │
├──────────────────────────────────────────────────────────────────────────────┤
│                                                                               │
│  Content Start X: -pageWidth/2 + marginLeft                                  │
│  Content Start Y:  pageHeight/2 - marginTop                                  │
│                                                                               │
│  Text Y Position: startY - (lineNumber × fontSize × lineHeight)              │
│  Text X Position: startX + (columnPosition × fontSize × 0.6)                 │
│                                                                               │
│  Heading Size:    fontSize × headingScale[elementType]                       │
│                                                                               │
│  Para Y Offset:   floor(lineNum / wordsPerLine) × paragraphSpacing          │
│  First Line X:    startX + indentFirstLine (if columnPosition === 0)         │
│                                                                               │
└──────────────────────────────────────────────────────────────────────────────┘

┌──────────────────────────────────────────────────────────────────────────────┐
│                    🔧 JavaScript API Quick Start                             │
├──────────────────────────────────────────────────────────────────────────────┤
│                                                                               │
│  // Initialize                                                                │
│  formatEngine = new WordWeaverFormatEngine(scene, camera);                   │
│                                                                               │
│  // Select format                                                             │
│  selectFormatProfile('word'); // or 'google', 'blog', 'pdf', 'manuscript'   │
│                                                                               │
│  // Create page with margins                                                 │
│  create3DDocumentPage(1, 0);                                                 │
│                                                                               │
│  // Custom preferences                                                        │
│  formatEngine.applyWriterPreferences({                                       │
│    fontSize: 0.5,                                                            │
│    lineHeight: 1.8,                                                          │
│    paragraphSpacing: 1.0                                                     │
│  });                                                                          │
│                                                                               │
│  // Calculate text position                                                  │
│  const pos = formatEngine.calculateTextPosition(lineNum, colPos, 'h1');     │
│                                                                               │
│  // Create 3D text                                                            │
│  await formatEngine.create3DText(text, 'body', position);                    │
│                                                                               │
│  // Save/Load                                                                 │
│  saveFormatPreferences();   // To localStorage                               │
│  loadFormatPreferences();   // From localStorage                             │
